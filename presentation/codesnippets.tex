\defverbatim[colored]\HaskellSTCode{
\begin{haskellcode}
  main :: IO ()
  main = do a
            print 2

  a, b :: IO ()
  a = do b
         print 20

  b = do print (crashSelf 2)
         print 200

  crashSelf :: Int -> Int
  crashSelf 0 = 1 `div` 0
  crashSelf x = crashSelf (x - 1)
\end{haskellcode}
}
